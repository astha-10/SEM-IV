def count_digits_letters(input_string):
    digit_count = 0
    letter_count = 0
    for char in input_string:
        if char.isdigit():
            digit_count += 1
        elif char.isalpha():
            letter_count += 1
    return digit_count, letter_count

# Example usage
input_string = "Hello, 123!"
digits, letters = count_digits_letters(input_string)
print("Number of digits:", digits)
print("Number of letters:", letters)


OUTPUT:
Number of digits: 3
Number of letters: 5
