def gcd(a, b):
    while b:
        a, b = b, a % b
    return a

# Example usage
num1 = 12
num2 = 18
print(f"GCD of {num1} and {num2} is:", gcd(num1, num2))

num1 = 48
num2 = 60
print(f"GCD of {num1} and {num2} is:", gcd(num1, num2))


OUTPUT:
GCD of 12 and 18 is: 6
GCD of 48 and 60 is: 12
