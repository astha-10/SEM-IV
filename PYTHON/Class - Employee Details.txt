class Employee:
    def __init__(self, emp_id, emp_name, emp_salary, emp_department):
        self.emp_id = emp_id
        self.emp_name = emp_name
        self.emp_salary = emp_salary
        self.emp_department = emp_department

    def calculate_emp_salary(self):
        # Add any logic here to calculate employee salary
        return self.emp_salary

    def emp_assign_department(self, department):
        self.emp_department = department

    def print_employee_details(self):
        print("Employee ID:", self.emp_id)
        print("Employee Name:", self.emp_name)
        print("Employee Salary:", self.emp_salary)
        print("Employee Department:", self.emp_department)

# Example usage
emp = Employee(emp_id=1, emp_name="John Doe", emp_salary=50000, emp_department="IT")
emp.print_employee_details()
print("Calculated Employee Salary:", emp.calculate_emp_salary())
emp.emp_assign_department("HR")
emp.print_employee_details()


OUTPUT:
Employee ID: 1
Employee Name: John Doe
Employee Salary: 50000
Employee Department: IT
Calculated Employee Salary: 50000
