def union_intersection(list1, list2):
    set1 = set(list1)
    set2 = set(list2)
    union_result = set1.union(set2)
    intersection_result = set1.intersection(set2)
    return list(union_result), list(intersection_result)

# Example usage
list1 = [1, 2, 3, 4, 5]
list2 = [3, 4, 5, 6, 7]

union, intersection = union_intersection(list1, list2)
print("Union of the two lists:", union)
print("Intersection of the two lists:", intersection)


OUTPUT:
Union of the two lists: [1, 2, 3, 4, 5, 6, 7]
Intersection of the two lists: [3, 4, 5]
