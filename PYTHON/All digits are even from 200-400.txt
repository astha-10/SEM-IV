def are_all_digits_even(number):
    for digit_str in str(number):
        if int(digit_str) % 2 != 0:
            return False
    return True

numbers = [str(num) for num in range(100, 401) if are_all_digits_even(num)]
result = ", ".join(numbers)
print(result)
