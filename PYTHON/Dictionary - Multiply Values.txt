def multiply_dict_values(d):
    result = 1
    for value in d.values():
        result *= value
    return result

# Example dictionary
my_dict = {'a': 10, 'b': 20, 'c': 30}

# Multiply all values in the dictionary
result = multiply_dict_values(my_dict)
print("Multiplication of all values in the dictionary:", result)

OUTPUT:
Multiplication of all values in the dictionary: 6000
